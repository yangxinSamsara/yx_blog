# 扇形

<!-- 一个底色正圆 两个半圆旋转 -->
<div class="wrap">
  <div class="sx1 sx"></div>
  <div class="sx2 sx"></div>
</div>

<style>
  .wrap {
    position: absolute;
    width: 200px;
    height: 200px;
    left: 0;
    right: 0;
    top: 0;
    bottom: 0;
    margin: auto;
    border-radius: 100px;
    background: yellow;
  }
  .sx {
    position: absolute;
    width: 200px;
    height: 200px;
    border-radius: 100px;
    background: red;
    clip: rect(0, 100px, 200px, 0);
  }
  .sx1 {
    transform: rotate(30deg);
  }
  .sx2 {
    transform: rotate(150deg);
  }
</style>

# ♥ 心形

<div class="wrap">
  <div class="heart_l heart"></div>
  <div class="heart_r heart"></div>
</div>

<style>
  .wrap {
    margin: 50px auto;
    width: 200px;
    height: 200px;
    position: relative;
  }
  .heart {
    width: 100px;
    height: 60px;
    background: red;
    border-radius: 0 50% 50% 0;
    position: absolute;
  }
  .heart_l {
    top: 0;
    left: 0;
    transform: rotate(-135deg);
  }
  .heart_r {
    transform: rotate(-45deg);
    left: 29px;
    top: 0;
  }
  .coin {
    width: 100px;
    height: 100px;
    background-color: goldenrod;
    border-radius: 50%;
    animation: move 5s infinite;
  }
  @keyframes move {
    0% {
      transform: rotateY(0);
    }
    100% {
      transform: rotateY(360deg);
    }
  }
</style>
