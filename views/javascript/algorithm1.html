/**
 * 输入:
  [{a:1,b:2,c:3},{b:2,c:3,a:1},{d:2,c:2}]
  输出:
  [{a:1,b:2,c:3},{d:2,c:2}]
 */
const arr = [{ a: 1, b: 2, c: 3 }, { b: 2, c: 3, a: 1 }, { d: 2, c: 2 }]
// 把数组中每个对象的key排序生成新对象然后变成对象字符串
function obj2str(obj) {
  const newObj = {}
  Object.keys(obj).sort().map(key => {
    newObj[key] = obj[key]
  })
  return JSON.stringify(newObj)
}
// 利用set去重
function unique(arr) {
  let set = new Set()
  arr.forEach(item => {
    set.add(obj2str(item))
  });
  return [...set].map(item => JSON.parse(item))
}

console.log(unique(arr));

# 反转字符串中的单词

// 输入: " the sky  is blue "
// 输出: "blue is sky the"

// const reverseWords = (words) => {
//   return words.trim().replace(/\s+/g, ' ').split(' ').reverse().join(' ')
// }

const reverseWords = (words) => {
  let left = 0, right = words.length - 1, queue = [], word = ''
  while (words.charAt(left) == ' ') {
    left++
  }
  while (words.charAt(right) == ' ') {
    right--
  }
  while (left <= right) {
    let char = words.charAt(left)
    // 当当前项为' '并且word有值时，往数组里面unshift，重置word
    if (char === ' ' && word) {
      queue.unshift(word)
      word = ''
    } else if (char !== ' ') {
      word += char
    }
    left++
  }
  queue.unshift(word)
  return queue.join(' ')
}
console.log(reverseWords(' the sky  is blue '));

const isPalindrome = (s) => {
  let left = 0, right = s.length - 1;
  while (left < right) {
    if (s.charAt(left) !== s.charAt(right)) {
      return false
    }
    left++
    right--
  }

}

  const arr = [...nums1, ...nums2].sort((a, b) => a - b)
  const mid = Math.ceil(arr.length / 2)

var twoSum = (nums, target) => {
  let map = new Map(), len = nums.length
  for (let i = 0; i < len; i++) {
    const diff = target - nums[i];
    if (!map.get(diff)) {
      map.set(nums[i], i)
    } else {
      return [map.get(diff), i]
    }
  }
}

const level = ["province", "city", "name"];
function transfrom(list, level) {
  const res = [];
  list.forEach(item => {
    pushItem(res, item, 0);
  });

  function pushItem(arr, obj, i) {
    const o = {
      value: obj[level[i]],
      children: [],
    };
    // 判断传入数组里是否有value等于要传入的项
    const hasItem = arr.find(el => el.value === obj[level[i]]);
    let nowArr;
    if(hasItem) {
      // 存在，则下一次遍历传入存在项的children
      nowArr = hasItem.children;
    }else{
      // 不存在 压入arr，下一次遍历传入此项的children
      arr.push(o);
      nowArr = o.children;
    }
    if(i === level.length - 1) delete o.children;
    i++;
    if(i < level.length) {
      // 递归进行层级的遍历
      pushItem(nowArr, obj, i);
    }
  }
  console.log(res);
}

function add(a ,b){
  let maxLength = Math.max(a.length, b.length);
  a = a.padStart(maxLength,'0')
  b = b.padStart(maxLength,'0')
  let res = '';
  let sum = 0;
  let ge = 0;
  let shi = 0;
  for (let i = maxLength - 1; i >=0; i--) {
    sum = parseInt(a[i]) + parseInt(b[i]) + shi
    shi = Math.floor(sum/10)
    ge = sum % 10
    res = ge + res
  }
  return res;
}

# 千分位

function format(num) {
  // 转化为字符串
  let str = num + "";
  // 以小数点分割为两部分
  let [int, dec = ""] = str.split(".");
  // 整数部分从后往前隔3个加，小数部分从前往后隔3个加，
  let intStr = "";
  for (let i = int.length - 1; i >= 0; i--) {
    if ((int.length - i) % 3 === 0 && i !== 0) {
      intStr = "," + int[i] + intStr;
    } else {
      intStr = int[i] + intStr;
    }
  }
  let decStr = "";
  if (dec.length > 0) {
    for (let i = 0; i < dec.length; i++) {
      let sum = decStr + dec[i];
      if ((i + 1) % 3 === 0) {
        decStr = sum + ",";
      } else {
        decStr = sum;
      }
    }
  }
  decStr = decStr.endsWith(',') ? decStr.slice(0,decStr.length - 1) : decStr
  return dec.length > 0 ? `${intStr}.${decStr}` : `${intStr}`;
}

console.log(format(100.123456789));

# 只出现一次的数字

// 给定一个非空整数数组，除了某个元素只出现一次以外，其余每个元素均出现两次。找出那个只出现了一次的元素。
// 任何数和自己做异或运算，结果为 0，即 a⊕a=0。
// 任何数和 0 做异或运算，结果还是自己，即 a⊕0=a。
// 异或运算中，满足交换律和结合律，也就是a⊕b⊕a=b⊕a⊕a=b⊕(a⊕a)=b⊕0=b。
const singleNumber = function(nums) {
  let init = nums[0];
  for(let i = 1; i < nums.length; i++){
      init ^=  nums[i];
  }
  return init;
};

var romanToInt = function(s) {
  const map = {
    I: 1,
    V: 5,
    IV: 4,
    IX: 9,
    X: 10,
    XL: 40,
    XC: 90,
    L: 50,
    C: 100,
    CD: 400,
    CM: 900,
    D: 500,
    M: 1000,
  };
  let res = 0;
  let index = 0;
  for (let i = 0; i < s.length; i++) {
    if (index + 1 < s.length && map[s.slice(index, index + 2)]) {
      res = res + map[s.slice(index, index + 2)];
      index += 2;
    } else {
      res = res + map[s.slice(index, index + 1)];
      index += 1;
    }
  }
  console.log(res);
  return res;
};

const generate = (rows = 5) => {
  if (rows === 0) {
    return [];
  }
  let res = Array.from(new Array(rows), () => []);

    res[i][0] = 1;
    res[i][i] = 1;

# 加一

// 给定一个由 整数 组成的 非空 数组所表示的非负整数，在该数的基础上加一。
// 最高位数字存放在数组的首位， 数组中每个元素只存储单个数字。
var plusOne = function(digits) {
  let step = 1;
  for (let i = digits.length - 1; i >= 0; i--) {
    let sum = 0;
    sum = digits[i] + step;
    digits[i] = sum % 10;
    step = (sum / 10) | 0;
  }
  if (digits[0] === 0) {
    digits.unshift(1);
  }
  console.log(digits);
};
plusOne([9, 9, 9]);//[1,0,0,0]

# x的平方根

var mySqrt = function(x) {
  let [l , r] = [0, x];
  let ans = -1;
  while (l <= r) {
    let mid = Math.floor((l+r)/2)
    if (mid**2 > x) {
      r = mid -1
    } else if(mid**2 < x){
      ans = mid//防止越界
      l = mid + 1
    }else{
      ans = mid
      return ans
    }
  }
  return ans;
};
const r = mySqrt(10)
